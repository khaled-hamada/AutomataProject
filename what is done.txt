<newline>            = \n | \r\n
<comment>            = // <character>* <newline>
<delimiter>          = { | [ | ( | } | ] | ) | : | ,|;
<operator>           = + | - | * | / | = | > | < | ! | & | % | ~ | $ | ^
<operator_extended>  = <operator> | <operator> = | && | ||

<letter>             = [a-zA-Z]

<identifier>         = (<letter> | _) (<letter> | <digit> | <operator> | _)*
